1. <input> 태그를 캘린더와 연결할 경우

	1.2. 이미지버튼을 이용하여 달력을 표시하는 경우

4. 날짜클릭시 유효성체크
	4.1. min, max속성설정
	//now, 혹은 today설정시 오늘날짜로 체크
	new BsCalendar('bsCal6', {min:'2012-05-01', max:'now'}).showCal();
	

	
	4.2. onClickValid에 체크함수 설정
	new BsCalendar('bsCal7', {
		//캘린더에서 함수로 dateStr, 캘린더를 파라미터로 전달
		onClickValid:function(dateStr, cal){
			if (dateStr > '2012-05-10') {
				alert('2012-05-10이전 날짜를 입력하세요.');
				return false;
			}
			
			return true;
		}
	}).showCal();
	


5. 년월변경시 유효성체크
	5.1. onChangeValid에 유효성체크 함수 설정
	function myCheck(year, month) {
		if (year < 2012) {
			alert('2012년 이후 날짜를 입력해 주십시오.');
			return false;
		}
		
		return true;
	}
	//익명함수로 직접기술하거나 기작성 함수의 함수명
	new BsCalendar('bsCal8', {onChangeValid:myCheck}).showCal();
